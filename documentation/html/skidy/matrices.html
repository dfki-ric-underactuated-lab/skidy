  7def generalized_vectors(
  8    DOF: int, startindex: int=0, derivatives: bool=False
  9    ) -> tuple[MutableDenseMatrix, MutableDenseMatrix, MutableDenseMatrix]:
 10    """Generate symbolic generalized vectors q, qd and q2d and optional
 11    q3d and q4d.
 12    
 13    The symbols are named as follows:
 14        
 15        q0, q1, ....., qi for joint positions.
 16        dq0, dq1, ....., dqi for joint velocities.
 17        ddq0, ddq1, ....., ddqi for joint accelerations.
 18        dddq0, dddq1, ....., dddqi for joint jerks.
 19        ddddq0, ddddq1, ....., ddddqi for joint jounces.
 20        
 21
 22    Args:
 23        DOF (int): Degrees of freedom.
 24        startindex (int, optional): Index of first joint. Defaults to 0.
 25        derivatives (bool, optional): return jerks and jounces too. 
 26            Defaults to False. 
 27
 28    Returns:
 29        tuple(sympy.Matrix): 
 30            Generalized vectors: (q,dq,d2q) if derivative == False
 31            else: (q,dq,d2q,q3d,q4d) 
 32    """
 33    if DOF > 1:
 34        q = Matrix(symbols(" ".join(f"q{i}" for i in range(startindex,startindex+DOF))))
 35        qd = Matrix(symbols(" ".join(f"dq{i}" for i in range(startindex,startindex+DOF))))
 36        q2d = Matrix(symbols(" ".join(f"ddq{i}" for i in range(startindex,startindex+DOF))))
 37        q3d = Matrix(symbols(" ".join(f"dddq{i}" for i in range(startindex,startindex+DOF))))
 38        q4d = Matrix(symbols(" ".join(f"ddddq{i}" for i in range(startindex,startindex+DOF))))
 39    else:
 40        q = Matrix([symbols(" ".join(f"q{i}" for i in range(startindex,startindex+DOF)))])
 41        qd = Matrix([symbols(" ".join(f"dq{i}" for i in range(startindex,startindex+DOF)))])
 42        q2d = Matrix([symbols(" ".join(f"ddq{i}" for i in range(startindex,startindex+DOF)))])
 43        q3d = Matrix([symbols(" ".join(f"dddq{i}" for i in range(startindex,startindex+DOF)))])
 44        q4d = Matrix([symbols(" ".join(f"ddddq{i}" for i in range(startindex,startindex+DOF)))])
 45    if derivatives:
 46        return q, qd, q2d, q3d, q4d
 47    else: 
 48        return q, qd, q2d

 50def joint_screw(axis: list, vec: list=[0,0,0], revolute: bool=True) -> MutableDenseMatrix:
 51    """Get joint screw coordinates from joint axis and vector to joint.
 52
 53    Args:
 54        axis (list): 
 55            Joint axis w.r.t. inertial frame.
 56        vec (list, optional): 
 57            Vector to joint axis from inertial frame for revolute joint. 
 58            Defaults to [0,0,0].
 59        revolute (bool, optional): 
 60            Revolute (True) or prismatic (False) joint. 
 61            Defaults to True.
 62
 63    Returns:
 64        sympy.Matrix: joint screw coordinates.
 65    """
 66    e = Matrix(axis)
 67        
 68    if revolute:
 69        y = Matrix(vec)
 70        return Matrix([e, y.cross(e)])
 71    else:
 72        return Matrix([0,0,0,*e])

 75def symbolic_inertia_matrix(
 76    index: int | str="", pointmass: bool=False) -> MutableDenseMatrix:
 77    """Create 3 x 3 symbolic inertia matrix with auto generated variable names.
 78
 79    Args:
 80        index (int or str): 
 81            postfix for variable name. Defaults to "".
 82        pointmass (bool): 
 83            Inertial matrix = I * Identity. Default to False.
 84    
 85    Returns:
 86        sympy.Matrix: Inertia matrix (3,3)
 87    """
 88    if pointmass:
 89        I = symbols(f"I{index}", real=1, constant=1)
 90        return I*Identity(3)
 91    Ixx, Iyy, Izz, Ixy, Ixz, Iyz = symbols(
 92        f"Ixx{index} Iyy{index} Izz{index} Ixy{index} Ixz{index} Iyz{index}")
 93    
 94    I = Matrix([[Ixx, Ixy, Ixz],
 95                [Ixy, Iyy, Iyz],
 96                [Ixz, Iyz, Izz]])
 97    return I

 99def SE3AdjInvMatrix(C: MutableDenseMatrix) -> MutableDenseMatrix:
100    """Compute Inverse of (6x6) Adjoint matrix for SE(3)
101
102    Args:
103        C (sympy.Matrix): SE(3) Pose.
104
105    Returns:
106        sympy.Matrix: Inverse of (6x6) Adjoint matrix
107    """
108    AdInv = Matrix([[C[0, 0], C[1, 0], C[2, 0], 0, 0, 0],
109                    [C[0, 1], C[1, 1], C[2, 1], 0, 0, 0],
110                    [C[0, 2], C[1, 2], C[2, 2], 0, 0, 0],
111                    [-C[2, 3]*C[1, 0]+C[1, 3]*C[2, 0], 
112                        C[2, 3]*C[0, 0]-C[0, 3]*C[2, 0],
113                        (-C[1, 3])*C[0, 0]+C[0, 3]*C[1, 0], 
114                        C[0, 0], C[1, 0], C[2, 0]],
115                    [-C[2, 3]*C[1, 1]+C[1, 3]*C[2, 1], 
116                        C[2, 3]*C[0, 1]-C[0, 3]*C[2, 1],
117                        (-C[1, 3])*C[0, 1]+C[0, 3]*C[1, 1], 
118                        C[0, 1], C[1, 1], C[2, 1]],
119                    [-C[2, 3]*C[1, 2]+C[1, 3]*C[2, 2], 
120                        C[2, 3]*C[0, 2]-C[0, 3]*C[2, 2],
121                        (-C[1, 3])*C[0, 2]+C[0, 3]*C[1, 2], 
122                        C[0, 2], C[1, 2], C[2, 2]]])
123    return AdInv

125def SE3AdjMatrix(C: MutableDenseMatrix) -> MutableDenseMatrix:
126    """Compute (6x6) Adjoint matrix for SE(3)
127
128    Args:
129        C (sympy.Matrix): SE(3) Pose.
130
131    Returns:
132    sympy.Matrix: (6x6) Adjoint matrix
133    """
134    Ad = Matrix([[C[0, 0], C[0, 1], C[0, 2], 0, 0, 0],
135                    [C[1, 0], C[1, 1], C[1, 2], 0, 0, 0],
136                    [C[2, 0], C[2, 1], C[2, 2], 0, 0, 0],
137                    [-C[2, 3]*C[1, 0]+C[1, 3]*C[2, 0], 
138                    -C[2, 3]*C[1, 1]+C[1, 3]*C[2, 1],
139                    -C[2, 3]*C[1, 2]+C[1, 3]*C[2, 2], 
140                    C[0, 0], C[0, 1], C[0, 2]],
141                    [C[2, 3]*C[0, 0]-C[0, 3]*C[2, 0],  
142                    C[2, 3]*C[0, 1]-C[0, 3]*C[2, 1],
143                    C[2, 3]*C[0, 2]-C[0, 3]*C[2, 2], 
144                    C[1, 0], C[1, 1], C[1, 2]],
145                    [-C[1, 3]*C[0, 0]+C[0, 3]*C[1, 0], 
146                    -C[1, 3]*C[0, 1]+C[0, 3]*C[1, 1],
147                    -C[1, 3]*C[0, 2]+C[0, 3]*C[1, 2], 
148                    C[2, 0], C[2, 1], C[2, 2]]])
149    return Ad

151def SE3adMatrix(X: MutableDenseMatrix) -> MutableDenseMatrix:
152    """Compute (6x6) adjoint matrix for SE(3) 
153        - also known as spatial cross product in the literature.
154
155    Args:
156        X (sympy.Matrix): (6x1) spatial vector.
157
158    Returns:
159        sympy.Matrix: (6x6) adjoint matrix
160    """
161    ad = Matrix([[0, -X[2, 0], X[1, 0], 0, 0, 0],
162                 [X[2, 0], 0, -X[0, 0], 0, 0, 0],
163                 [-X[1, 0], X[0, 0], 0, 0, 0, 0],
164                 [0, -X[5, 0], X[4, 0], 0, -X[2, 0], X[1, 0]],
165                 [X[5, 0], 0, -X[3, 0], X[2, 0], 0, -X[0, 0]],
166                 [-X[4, 0], X[3, 0], 0, -X[1, 0], X[0, 0], 0]])
167    return ad

169def SE3Exp(XX: MutableDenseMatrix, t: float | Expr) -> MutableDenseMatrix:
170    """compute exponential mapping for SE(3).
171
172    Args:
173        XX (sympy.Matrix): (6,1) spatial vector.
174        t (sympy.Expr): rotation angle.
175
176    Returns:
177        sympy.Matrix: (4,4) SE(3) Pose.
178    """
179    X = XX.T
180    xi = Matrix(X[0:3])
181    eta = Matrix(X[3:6])
182    xihat = Matrix([[0, -X[2], X[1]],
183                    [X[2], 0, -X[0]],
184                    [-X[1], X[0], 0]])
185    R = Matrix(Identity(3)) + sin(t)*xihat + (1-cos(t))*(xihat*xihat)
186    if xi == zeros(3, 1):
187        p = eta * t
188    else:
189        p = (Matrix(Identity(3))-R)*(xihat*eta) + xi*(xi.T*eta)*t
190    C = R.row_join(p).col_join(Matrix([0, 0, 0, 1]).T)
191    return C

193def SE3Inv(C: MutableDenseMatrix) -> MutableDenseMatrix:
194    """Compute analytical inverse of exponential mapping for SE(3).
195
196    Args:
197        C (sympy.Matrix): (4,4) SE(3) Pose.
198
199    Returns:
200        sympy.Matrix: (4,4) Inverse of SE(3) Pose.
201    """
202    CInv = Matrix([[C[0, 0], C[1, 0], C[2, 0], 
203                    -C[0, 0]*C[0, 3]-C[1, 0]*C[1, 3]-C[2, 0]*C[2, 3]],
204                    [C[0, 1], C[1, 1], C[2, 1], 
205                    -C[0, 1]*C[0, 3]-C[1, 1]*C[1, 3]-C[2, 1]*C[2, 3]],
206                    [C[0, 2], C[1, 2], C[2, 2], -C[0, 2] *
207                        C[0, 3]-C[1, 2]*C[1, 3]-C[2, 2]*C[2, 3]],
208                    [0, 0, 0, 1]])
209    return CInv

211def SO3Exp(axis: MutableDenseMatrix, angle: float | Expr) -> MutableDenseMatrix:
212    """Compute exponential mapping for SO(3).
213
214    Args:
215        axis (sympy.Matrix): Rotation axis
216        angle (double): Rotation angle
217
218    Returns:
219        sympy.Matrix: Rotation matrix
220    """
221    axis = Matrix(axis)
222    xhat = Matrix([[0, -axis[2, 0], axis[1, 0]],
223                    [axis[2, 0], 0, -axis[0, 0]],
224                    [-axis[1, 0], axis[0, 0], 0]])
225    R = Matrix(Identity(3)) + sin(angle) * xhat + (1-cos(angle))*(xhat*xhat)
226    return R

249def transformation_matrix(r: MutableDenseMatrix=Matrix(Identity(3)), 
250                         t: MutableDenseMatrix | list=zeros(3, 1)) -> MutableDenseMatrix:
251    """Build transformation matrix from rotation and translation.
252
253    Args:
254        r (sympy.Matrix): 
255            SO(3) Rotation matrix (3,3). 
256            Defaults to sympy.Matrix(Identity(3))
257        t (sympy.Matrix | list): 
258            Translation vector (3,1). Defaults to sympy.zeros(3,1)
259
260    Returns:
261        sympy.Matrix: Transformation matrix (4,4)
262    """
263    if type(t) is list:
264        t = Matrix(t)
265    T = Matrix(Identity(4))
266    T[:3,:3] = r
267    T[:3,3] = t
268    return T

270def mass_matrix_mixed_data(m: float | Expr, Theta: MutableDenseMatrix, 
271                        COM: MutableDenseMatrix) -> MutableDenseMatrix:
272    """Build mass-inertia matrix in SE(3) from mass, inertia and 
273    center of mass information.
274
275    Args:
276        m (float | sympy.Expr): Mass.
277        Theta (array_like): Inertia (3,3).
278        COM (array_like): Center of mass (3,1).
279
280    Returns:
281        sympy.Matrix: Mass-inertia matrix (6,6).
282    """
283    M = Matrix([[Theta[0, 0], Theta[0, 1], Theta[0, 2], 0, 
284                    (-COM[2])*m, COM[1]*m],
285                [Theta[0, 1], Theta[1, 1], Theta[1, 2],
286                    COM[2]*m, 0, (-COM[0]*m)],
287                [Theta[0, 2], Theta[1, 2], Theta[2, 2],
288                    (-COM[1])*m, COM[0]*m, 0],
289                [0, COM[2]*m, (-COM[1]*m), m, 0, 0],
290                [(-COM[2])*m, 0, COM[0]*m, 0, m, 0],
291                [COM[1]*m, (-COM[0])*m, 0, 0, 0, m]])
292    return M

294def mass_matrix_mixed_data_identification(m: float | Expr, Theta: MutableDenseMatrix, 
295                        h: MutableDenseMatrix) -> MutableDenseMatrix:
296    """Build mass-inertia matrix in SE(3) from mass, inertia and 
297    first moment of mass, i.e. m*com_vector.

299    Args:
300        m (float): Mass.
301        Theta (array_like): Inertia (3,3).

305        sympy.Matrix: Mass-inertia matrix (6,6).

313                [0, h[2], (-h[1]), m, 0, 0],
314                [(-h[2]), 0, h[0], 0, m, 0],
315                [h[1], (-h[0]), 0, 0, 0, m]])
316    return M

336def rpy_to_matrix(coords: list | MutableDenseMatrix) -> MutableDenseMatrix:
337    """Convert roll-pitch-yaw coordinates to a 3x3 homogenous rotation matrix.
338
339    The roll-pitch-yaw axes in a typical URDF are defined as a
340    rotation of ``r`` radians around the x-axis followed by a rotation of
341    ``p`` radians around the y-axis followed by a rotation of ``y`` radians
342    around the z-axis. These are the Z1-Y2-X3 Tait-Bryan angles. See
343    Wikipedia_ for more information.
344
345    .. _Wikipedia: https://en.wikipedia.org/wiki/Euler_angles#Rotation_matrix
346
347    Args:
348        coords (MutableDenseMatrix | list): 
349            The (3,) roll-pitch-yaw coordinates in order (x-rot, y-rot, z-rot).
350
351    Returns:
352        MutableDenseMatrix: 
353            The corresponding homogenous 3x3 rotation matrix.        
354    """
355    c3 = cos(coords[0])
356    c2 = cos(coords[1])
357    c1 = cos(coords[2])
358    s3 = sin(coords[0])
359    s2 = sin(coords[1])
360    s1 = sin(coords[2])
361    return Matrix([
362        [c1 * c2, (c1 * s2 * s3) - (c3 * s1), (s1 * s3) + (c1 * c3 * s2)],
363        [c2 * s1, (c1 * c3) + (s1 * s2 * s3), (c3 * s1 * s2) - (c1 * s3)],
364        [-s2, c2 * s3, c2 * c3]
365    ])

383def matrix_to_rpy(R: ArrayLike, solution: int=1) -> np.ndarray:
384    """Convert a 3x3 transform matrix to roll-pitch-yaw coordinates.
385
386    The roll-pitchRyaw axes in a typical URDF are defined as a
387    rotation of ``r`` radians around the x-axis followed by a rotation of
388    ``p`` radians around the y-axis followed by a rotation of ``y`` radians
389    around the z-axis. These are the Z1-Y2-X3 Tait-Bryan angles. See
390    Wikipedia_ for more information.
391
392    .. _Wikipedia: https://en.wikipedia.org/wiki/Euler_angles#Rotation_matrix
393
394    There are typically two possible roll-pitch-yaw coordinates that could have
395    created a given rotation matrix. Specify ``solution=1`` for the first one
396    and ``solution=2`` for the second one.
397
398    Args:
399        R (ArrayLike): 
400            A 3x3 homogenous rotation matrix.
401        solution (int, optional): 
402            Either 1 or 2, indicating which solution to return. 
403            Defaults to 1.
404
405    Returns:
406        ndarray:
407            The (3,) roll-pitch-yaw coordinates in order (x-rot, y-rot, z-rot).
408    """
409    R = np.asanyarray(R, dtype=np.float64)
410    r = 0.0
411    p = 0.0
412    y = 0.0
413
414    if np.abs(R[2,0]) >= 1.0 - 1e-12:
415        y = 0.0
416        if R[2,0] < 0:
417            p = np.pi / 2
418            r = np.arctan2(R[0,1], R[0,2])
419        else:
420            p = -np.pi / 2
421            r = np.arctan2(-R[0,1], -R[0,2])
422    else:
423        if solution == 1:
424            p = -np.arcsin(R[2,0])
425        else:
426            p = np.pi + np.arcsin(R[2,0])
427        r = np.arctan2(R[2,1] / np.cos(p), R[2,2] / np.cos(p))
428        y = np.arctan2(R[1,0] / np.cos(p), R[0,0] / np.cos(p))
429
430    return np.array([r, p, y], dtype=np.float64)

447def quaternion_to_matrix(Q: list | MutableDenseMatrix) -> MutableDenseMatrix:
448    """Convert a quaternion into SO(3) rotation matrix.
449
450    Args:
451        Q (list | sympy.MutableDenseMatrix): Quaternion in order [w,x,y,z].
452        
453    Returns:
454        sympy.Matrix: (3,3) Rotation matrix.
455    """
456    # ensure symbolic values
457    Q = Matrix(Q)  
458    
459    # Extract the values from Q
460    q0 = Q[0]
461    q1 = Q[1]
462    q2 = Q[2]
463    q3 = Q[3]
464
465    # First row of the rotation matrix
466    r00 = 2 * (q0 * q0 + q1 * q1) - 1
467    r01 = 2 * (q1 * q2 - q0 * q3)
468    r02 = 2 * (q1 * q3 + q0 * q2)
469
470    # Second row of the rotation matrix
471    r10 = 2 * (q1 * q2 + q0 * q3)
472    r11 = 2 * (q0 * q0 + q2 * q2) - 1
473    r12 = 2 * (q2 * q3 - q0 * q1)
474
475    # Third row of the rotation matrix
476    r20 = 2 * (q1 * q3 - q0 * q2)
477    r21 = 2 * (q2 * q3 + q0 * q1)
478    r22 = 2 * (q0 * q0 + q3 * q3) - 1
479
480    # 3x3 rotation matrix
481    rot_matrix = Matrix([[r00, r01, r02],
482                         [r10, r11, r12],
483                         [r20, r21, r22]])
484
485    return rot_matrix

15        
16        q0, q1, ....., qi for joint positions.

19        dddq0, dddq1, ....., dddqi for joint jerks.

25        startindex (int, optional): Index of first joint. Defaults to 0.
26        derivatives (bool, optional): return jerks and jounces too. 
27            Defaults to False. 
28
29    Returns:
30        tuple(sympy.Matrix): 
31            Generalized vectors: (q,dq,d2q) if derivative == False
32            else: (q,dq,d2q,q3d,q4d)

171    """compute exponential mapping for SE(3).

414

450

453        
454    Returns:
455        sympy.Matrix: (3,3) Rotation matrix.
456    """
457    # ensure symbolic values
458    Q = Matrix(Q)  
459    
460    # Extract the values from Q
461    q0 = Q[0]
462    q1 = Q[1]
463    q2 = Q[2]
464    q3 = Q[3]
465
466    # First row of the rotation matrix
467    r00 = 2 * (q0 * q0 + q1 * q1) - 1
468    r01 = 2 * (q1 * q2 - q0 * q3)
469    r02 = 2 * (q1 * q3 + q0 * q2)
470
471    # Second row of the rotation matrix
472    r10 = 2 * (q1 * q2 + q0 * q3)
473    r11 = 2 * (q0 * q0 + q2 * q2) - 1
474    r12 = 2 * (q2 * q3 - q0 * q1)
475
476    # Third row of the rotation matrix
477    r20 = 2 * (q1 * q3 - q0 * q2)
478    r21 = 2 * (q2 * q3 + q0 * q1)
479    r22 = 2 * (q0 * q0 + q3 * q3) - 1
480
481    # 3x3 rotation matrix
482    rot_matrix = Matrix([[r00, r01, r02],
483                         [r10, r11, r12],
484                         [r20, r21, r22]])
485
486    return rot_matrix
